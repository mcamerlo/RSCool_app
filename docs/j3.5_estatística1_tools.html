Plot com intervalo de confiança

mod <- lm(var_numy ~ var_numx, data = D)
# predicts + interval
newx <- seq(min(D$var_numx), max(D$var_numx), length.out=100)
preds <- predict(mod, newdata = data.frame(var_numx=newx), 
                 interval = 'confidence')

# plot
plot(D$var_numy ~ D$var_numx, data = D)
# add fill
polygon(c(rev(newx), newx), c(rev(preds[ ,3]), preds[ ,2]), border = NA)
# model
abline(mod)
# intervals
lines(newx, preds[ ,3], lty = 'dashed', col = 'red')
lines(newx, preds[ ,2], lty = 'dashed', col = 'red')
